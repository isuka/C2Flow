
// return, exitでフローが切れる
func1 {
    if (cond1) {
        nop1
        return
    } else if (cond2) {
        nop2
        exit
    }

    nop3
}

func2 {
    switch (cond1) {
    case fuga
        nop1
        return
    case piyo
        nop2
        break
    default
        nop3
        exit 1
    }
    return 0
}

}
